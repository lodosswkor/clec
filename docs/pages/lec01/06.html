자기소개 하기

#include <stdio.h>

int main() {

    char name[80]; 
    char home[80];
    int age; 

    printf("이름이 뭐예요? : "); 
    scanf("%s", name); 
    printf("어디 살아요? : "); 
    scanf("%s", home); 
    printf("나이가 어케 되시는지... : "); 
    scanf("%d", &age); 

    printf("==============================\n"); 
    printf("이 름 : %s\n", name); 
    printf("서식지 : %s\n", home); 
    printf("나 이 : %d\n", age); 
    printf("==============================\n"); 


    return 0; 
}

간단한 주사위 게임

#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main() {

    int score_player_a = 0;
    int score_player_b = 0; 
    int max_gamecnt = 2; 
    int dice = 0 ;
    char rtn = 0; 

    printf("게임 설명 : 주사위를 총 %d 번 굴려 합친 주사위 눈의 합이 더 높은 사람이 이기게 됩니다.", max_gamecnt);
    printf("\n주사위 게임을 시작하겠습니다.\n");

    srand(time(NULL)); 

    while(max_gamecnt > 0) {

        printf("\nPlayer A 주사위를 던져주세요. [enter]");

        rtn = getchar();

        dice = (rand() % 6) + 1;
        score_player_a += dice; 

        printf("Player A 점수 : %d\n", dice);

        printf("\nPlayer B 주사위를 던져주세요. [enter]");

        rtn = getchar(); 

        dice = (rand() % 6) + 1;
        score_player_b += dice; 

        printf("Player B 점수 : %d\n", dice);  

        max_gamecnt --; 
    }  

    printf("\n========================================\n");
        printf("Player A 총합: %d\n", score_player_a);
        printf("Player B 총합: %d\n", score_player_b);
    printf("========================================\n");



}

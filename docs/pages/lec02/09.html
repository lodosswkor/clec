표준입출력 stdio.h

표준입출력 stdio.h

  • 출력 (stdout)
    • putchar()
    • puts()
    • printf()
  • - 입력 (stdin)
    • getchar()
    • gets()
    • scanf()

출력 (stdout)

putchar()

  • 한문자를 출력한다.
#include <stdio.h>
int main() {
    putchar('db');
    return 0; 
}

puts()

  • 문자열 (char 배열/포인터)를 출력한다. 
#include <stdio.h>
int main() {
    puts("Hello World!");
    return 0; 
}

printf()

#include <stdio.h>
int main() {
    printf('나의 이름은 %s 입니다.', "이종석");
    return 0; 
}

- 입력 (stdin)

getchar()

  • 한문자를 읽어들인다.
#include <stdio.h>

int main() {
    char ch = 0; 
    printf("문자를 입력하세요. : ");
    ch = getchar(); 
    printf("문자 : %c \n", ch);
    return 0; 
}

gets()

  • 문자열을 입력받는다. 
#include <stdio.h>

int main() {
    char input[256];
    printf("문자열을 입력하세요. : ");
    gets(input);
    printf("문자열 : %s \n", input);
    return 0;
}

scanf()

//--- MS C에서만 사용 ---------------
#ifdef _MSC_VER
 #define _CRT_SECURE_NO_WARNINGS
#endif 
//--------------------------------

#include <stdio.h>

int main() {

    char input[256];
    printf("문자열을 입력하세요. : ");
    scanf("%s", input);
    printf("문자열 : %s ", input);
    return 0;
}

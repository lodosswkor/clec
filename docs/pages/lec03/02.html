동등연산자

동등연산자 ( ==, != )

기호 의미
== (좌항이 우항과) 같다.
!= 같지않다 
#include <stdio.h>

int main() {

    char a = 'a'; 
    char b = 'b';

    printf("char a의 값 %c(ASCII %d)는 char b의 값 %c(ASCII %d)와 같습니다. %d\n", a, a, b, b, a == b); 
    printf("char a의 값 %c(ASCII %d)는 char b의 값 %c(ASCII %d)와 다릅니다. %d\n", a, a, b, b, a != b); 

    return 0; 
}

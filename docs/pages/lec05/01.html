함수의 기본 구조

반환 자료형 함수명 ( 인수 ) {
  함수 몸체 
}

반환값이 있는 함수

#include <stdio.h>

// 예제 
int add( int x, int y) {
    return x + y; 
}

int main() {
    int x = 10, y = 1;
    int val = add(x, y); 
    printf("%d + %d = %d 입니다.\n", x, y, val);
    return 0; 
}

반환값이 없는 함수

#include <stdio.h>

void printAdd( int x, int y) {
    int val = x + y; 
    printf("%d + %d = %d 입니다.\n", x, y, val);
}

int main() {
    printAdd( 10, 1);  
    return 0; 
}

인자가 있는 함수

#include <stdio.h>

void divide( int x, int y) {
    int val = x / y; 
    return val; 
}

int main() {
    int x = 10, y = 1;
    printf("%d + %d = %d 입니다.\n", x, y, val);
    return 0; 
}

인자가 없는 함수

#include <stdio.h>

void printMessage() {
    printf("인자가 없는 함수입니다.");
}

int main() {
    int x = 10, y = 1;
    printf("%d + %d = %d 입니다.\n", x, y, val);
    return 0; 
}

정적함수

#include <stdio.h>

static void printAdd( int x, int y) {
    int val = x + y; 
    printf("%d + %d = %d 입니다.\n", x, y, val);
}

int main() {
    printAdd( 10, 1);  
    return 0; 
}

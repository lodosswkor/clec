### 변수와 메모리 할당 위치 

```c
#include <stdio.h>

int a = 100; // DATA 영역 
int b = 100; // DATA 영역 

void funcA();

int main() {
    int a = 10; // STACK 영역 
    funcA();
    printf("%d\n", a);  
}

void funcA() {
    int a = 20; // STACK 영역
    printf("funcA() 호출");
    return; 
}

```

### 일반 변수, char[], char* 

```c
#include <stdio.h>

int a = 100; // DATA 영역 
int b = 100; // DATA 영역 

int main() {
    int a = 10; // STACK 영역
    char arr[10] = "hello"; 
    char *str = "hello"; 

    arr = "hello2"; // error 
    strcpy(arr, "hello2");

    str = "hello2"; 
    strcpy(str, "hello2"); // error

    printf("%d\n", a);  
}

```

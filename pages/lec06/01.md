# 메모리의 구조와 포인터  
---


### 메모리 영역

```c
#include <stdio.h>

int a = 1;  // 전역 변수

int main(void)
{
    int num1 = 10, num2 = 20;  // 지역 변수
    static int s = 20;  // 정적 변수

    printf("데이터 출력 : %d %d %d %d\n", a, num1, num2, s);
    printf("코드 영역 : %p \t %p\n", main, printf);
    printf("스텍 영역 : %p \t %p\n", &num1, &num2);
    printf("데이터 영역 : %p \t %p\n", &a, &s);

    return 0;
}
```
